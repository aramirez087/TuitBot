### Task 1: TypeScript Types and API Wiring

**Types to add in `types.ts`:**

```typescript
// --- Mined Angle types (Hook Miner) ---

export interface EvidenceItemResponse {
  evidence_type: string;    // 'contradiction' | 'data_point' | 'aha_moment'
  citation_text: string;    // max 120 chars
  source_node_id: number;
  source_note_title: string;
  source_heading_path?: string;
}

export interface MinedAngle {
  angle_type: string;       // 'story' | 'listicle' | 'hot_take'
  seed_text: string;        // the hook text
  char_count: number;
  evidence: EvidenceItemResponse[];
  confidence: string;       // 'high' | 'medium'
  rationale: string;        // 1-sentence explanation
}

export interface AssistAnglesResponse {
  angles: MinedAngle[];
  fallback_reason?: string; // null when angles are present
  topic: string;
  vault_citations?: VaultCitation[];
}
```

**Client method in `client.ts`:**

Add `angles` to the `assist` namespace:

```typescript
angles: (
  topic: string,
  acceptedNeighborIds: number[],
  opts?: { sessionId?: string; selectedNodeIds?: number[] }
) =>
  request<AssistAnglesResponse>('/api/assist/angles', {
    method: 'POST',
    body: JSON.stringify({
      topic,
      accepted_neighbor_ids: acceptedNeighborIds,
      ...(opts?.sessionId && { session_id: opts.sessionId }),
      ...(opts?.selectedNodeIds && { selected_node_ids: opts.selectedNodeIds }),
    })
  }),
```

**Design decisions:**
- D20: `accepted_neighbor_ids` is a required positional parameter (not optional) because the Rust endpoint returns 400 if empty. The caller must check conditions before calling.
- D21: Response types use `string` for `angle_type` / `evidence_type` / `confidence` (not enums) to match the flexible backend and avoid breaking on new values. Display logic lives in `angleStyles.ts`.

### Task 5: VaultSelectionReview Integration

This is the core wiring change. The `handleGenerate` function needs branching logic.

**Current flow:**
```
handleGenerate() → api.assist.hooks() → hookOptions → HookPicker
```

**New flow:**
```
handleGenerate()
  ├── synthesisEnabled && acceptedNeighbors.size > 0?
  │     YES → api.assist.angles() → angleResult
  │            ├── angles present? → AngleCards
  │            └── fallback_reason? → AngleFallback
  └── NO → api.assist.hooks() → hookOptions → HookPicker (unchanged)
```

**New state variables:**
```typescript
let angleResult = $state<AssistAnglesResponse | null>(null);
let angleLoading = $state(false);
let angleError = $state<string | null>(null);
let showAngleFallback = $state(false);
let angleFallbackReason = $state<string | null>(null);
```

**Modified `handleGenerate`:**
```typescript
async function handleGenerate() {
  if (!selection || hookLoading || angleLoading) return;

  const topic = selection.selected_text || selection.note_title || selection.heading_context || 'general topic';

  // Condition: synthesis + neighbors → mine angles
  if (synthesisEnabled && acceptedNeighbors.size > 0) {
    angleLoading = true;
    angleError = null;
    try {
      const neighborIds = [...acceptedNeighbors.keys()];
      const result = await api.assist.angles(topic, neighborIds, { sessionId });
      if (result.fallback_reason || result.angles.length === 0) {
        // Fallback state
        showAngleFallback = true;
        angleFallbackReason = result.fallback_reason || 'all_angles_filtered';
        trackAngleFallback(result.fallback_reason ?? 'empty', acceptedNeighbors.size, sessionId);
      } else {
        angleResult = result;
        trackAnglesMined(acceptedNeighbors.size, result.angles.length, sessionId);
      }
    } catch (e) {
      if (e instanceof Error && e.message.includes('timeout')) {
        showAngleFallback = true;
        angleFallbackReason = 'timeout';
      } else {
        angleError = e instanceof Error ? e.message : 'Failed to mine angles';
      }
    } finally {
      angleLoading = false;
    }
  } else {
    // Existing generic hooks path (unchanged)
    hookLoading = true;
    hookError = null;
    error = null;
    try {
      const result = await api.assist.hooks(topic, { sessionId });
      hookOptions = result.hooks;
      trackHooksGenerated(acceptedNeighbors.size, graphNeighbors.length, result.hooks.length, sessionId);
    } catch (e) {
      hookError = e instanceof Error ? e.message : 'Failed to generate hooks';
    } finally {
      hookLoading = false;
    }
  }
}
```

**New handler: `handleAngleSelected`:**
```typescript
async function handleAngleSelected(angle: MinedAngle, format: 'tweet' | 'thread') {
  if (!selection || generating) return;
  if (hasExistingContent && !confirmReplace) {
    confirmReplace = true;
    return;
  }
  generating = true;
  error = null;
  confirmReplace = false;
  try {
    const nodeIds = selection.resolved_node_id ? [selection.resolved_node_id] : [];
    const neighborProv: Array<{ node_id: number; edge_type?: string; edge_label?: string }> = [];
    for (const [nid, { neighbor }] of acceptedNeighbors) {
      if (!nodeIds.includes(nid)) nodeIds.push(nid);
      neighborProv.push({
        node_id: nid,
        edge_type: neighbor.reason,
        edge_label: neighbor.reason_label,
      });
    }
    // Pass seed_text as highlight, angle_type as hookStyle
    await ongenerate(nodeIds, format, [angle.seed_text], angle.angle_type, neighborProv.length > 0 ? neighborProv : undefined);
  } catch (e) {
    error = e instanceof Error ? e.message : 'Generation failed';
  } finally {
    generating = false;
  }
}
```

## Design Decisions

| ID | Decision | Rationale |
|----|----------|-----------|
| D20 | `acceptedNeighborIds` is a required positional param in `api.assist.angles()` | Matches Rust endpoint contract (returns 400 if empty). Caller must guard. |
| D21 | Response types use `string` not TS enums for angle/evidence types | Flexible — new types from backend don't break frontend. Display logic centralized in `angleStyles.ts`. |
| D22 | Separate `angleStyles.ts` from `hookStyles.ts` | Different domains (mined angles vs generic hooks). Shared `getConfidenceBadge` is imported, not duplicated. |
| D23 | AngleFallback is a separate component (not inline in VaultSelectionReview) | Keeps VaultSelectionReview template manageable. Fallback has its own layout and two action buttons. |
| D24 | "More hook styles" transitions from AngleCards to HookPicker without losing neighbor state | acceptedNeighbors/dismissedNodeIds/synthesisEnabled are component-level state, not reset by view transitions. |
| D25 | From Vault chunk-selection path (`FromVaultPanel` without `selectionSessionId`) is NOT modified | UX spec section 2 explicitly excludes this path from angle mining (no neighbors = no evidence). |
| D26 | AngleCards shares HookPicker's format toggle and confirm button patterns exactly | Visual consistency. Users should recognize the interaction even though the content differs. |
| D27 | Client-side 15s timeout is NOT implemented in V1 | The standard `fetch` timeout depends on server response. A client-side AbortController timeout adds complexity. If latency is an issue, it can be added in a follow-up. The fallback path already handles errors gracefully. |
